This ontology is based on Provenance ontology developed by https://www.w3.org/TR/prov-o/
Squirrel predicates were introduced for special functions like hostedOn, numberOfTriples, status.

Example

@prefix rdf:  <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix rdfs:  <http://www.w3.org/2000/01/rdf-schema#> .
@prefix xsd:  <http://www.w3.org/2001/XMLSchema#> .
@prefix dcterms:   <http://purl.org/dc/terms/> .
@prefix owl:   <http://www.w3.org/2002/07/owl#> .
@prefix prov:   <http://www.w3.org/ns/prov-o/> .
@prefix:   <http://sample.com/> .

:crawlingActivity_1
   a prov:Activity;
   dcterms:description "crawling activity of a worker.";
   prov:wasAssociatedWith :Worker_1;
   prov:startedAtTime "2012-04-30T20:40:40"^^xsd:dateTime;
   prov:endedAtTime "2012-04-30T20:40:40"^^xsd:dateTime;
   prov:qualifiedAssociation [
     a prov:Association;
     prov:agent :worker_1;

     # The worker does the crawling activity.
     prov:hadPlan :crawling;
     ];
   sq:hostedOn "http://localhost:3030/"^^xsd:string;
   sq:numberOfTriples "123456"^^xsd:long;
   sq:Status "successful"^^xsd:string;
   .

   :worker_1
   a prov:Agent;
   .

   :crawling
   a prov:Plan, prov:Entity;
   rdfs:comment "steps: http fetcher, rdf analyzer, sink"@en;
   .

   :resultgraph
   a prov:Entity;
   prov:wasGeneratedBy :crawlingActivity_1;
   .
